# VuePress搭建Blog实战

# 一.写在开头

# 为什么写博客

  • 首先记录自己的所学，并加深理解（时间走远，记忆不在，这就是你逝去的青春和头发）。
  • 简历中个人网站，体现自己价值和能力（为了小钱钱）。
  • 温故而知新（寻找遗忘的记忆）。

# 为什么是Vuepress

  • 介绍请转官网，详见 VuePress中文网；
  • 容易上手（公司小伙伴推荐之后使用了下，哎哟，不错哟）；
  • 兼容性好，扩展markdown（最近个人的笔记都是这个写的）；
  • 为了之后进一步学习Vue。

# 二. 搭建

有些朋友不想看官网，想直接入手的话，我就不偷懒了。

# 安装弄得Node

下载转node.js，确保node的版本大于8(vuepress官网介绍)

#查看安装完成版本（本人是v12.14.0版本）
node -v 全局安装 VuePres
#NPM是随同NodeJS一起安装的包管理工具,查看npm的版本（本人是6.13.4版本）
npm -v

安装vuepress，详见官网

# 安装
npm install -g vuepress
# 进入项目目录，不存在就新建。
cd D:\INDEX_SHAO\vuepress
#初始化项目，npm init默认是yes
npm init -y

# 布局项目目录

vuepress
├─── docs
│   ├── README.md
│   └── .vuepress              #存放VuePress相关的文件
│       ├── components         #放静态资源文件，个人用于-js
│       ├── dist               #放静态资源文件，个人用于-img
│       ├── public             #放静态资源文件，个人用于-vue
│       └── config.js          #VuePress必要的配置文件
│   └── backend                   #一般文章目录
│   	├── README.md		   #内部实际文件
└── package.json               #项目配置文件和脚本执行

# 主要配置文件config.js

  • title：网站标题
  • description：网站描述
  • head：额外的需要被注入到当前页面的HTML"head"中的标签，其中路径的"/"就是public资源目录。
  • 具体配置详情请看文档：配置
  • PWD配置
module.exports = {
    # ↓↓↓↓↓↓↓↓网站信息配置↓↓↓↓↓↓
    title: '个人主页',
    description: '邵先生的博客',
    head: [
        ['link', { rel: 'icon', href: '/img/logo.ico' }],
        ['link', { rel: 'manifest', href: '/manifest.json' }],
        ['link', { rel: 'apple-touch-icon', href: '/img/logo.png' }],
        ['meta', { name: 'viewport', content: 'width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=0;' }],
    ],
    # ↓↓↓↓↓↓↓↓主题配置↓↓↓↓↓↓
    themeConfig: {
        #nav：导航栏配置，此配置主要用于配置导航栏的链接
        nav: [
            { text: '主页', link: '/' },
            {
                text: '博文',
                items: [
                    { text: '后端', link: '/backend/' },
                    { text: '前端', link: '/frontend/' },
                    { text: '工具', link: '/tools/' }
                ]
            },
            { text: '工具', link: '/ide/' },
            { text: 'Github', link: 'https://www.github.com/codeteenager' },
        ],
        #sidebar：侧边栏配置，你可以省略.md拓展名，同时以/结尾的路径将会被视为 */README.md
        sidebar: {
            '/backend/': [
                {
                    title: 'JAVA源码学习',
                    collapsable: false,
                    children: [
                        ['mybatis','mybatis学习'],
                        ['spring','spring学习']

                    ]
                },
                {
                    title: 'spring',
                    collapsable: false,
                    children: [
                        ['spring','spring']
                    ]
                }
            ],
            '/frontend/':[
                {
                    title: '所见即所得',
                    collapsable: false,
                    children: [
                        "",
                        ['vue','vue']
                    ]
                }
            ],
            '/tools/':[
                {
                    title: '必先利其器',
                    collapsable: false,
                    children: [
                        "",
                        ['Git操作指南','Git操作指南.md'],
                        ['markdown指南','markdown指南'],
                        ['idea','idea']
                    ]
                }
            ]
        },
        #嵌套的标题链接深度，默认的深度为1    
        sidebarDepth: 2,
        #最后更新时间
        lastUpdated: 'Last Updated',
    },
    # PWD配置，需要在基本配置中开启serviceWorke
    serviceWorker: true,
    ga: 'UA-120261845-1',
}

# 配置打包package.json

{
  "name": "vuepress",
  "version": "1.0.0",
  "main": "index.js",
  "directories": {
    "doc": "docs"
  },
  "scripts": {
    #启动运行
    "dev": "vuepress dev docs",
    #构建
    "build": "vuepress build docs",
    #执行脚本
    "deploy": "bash deploy.sh"
  },
  "keywords": [],
  "author": "shaominjie-index",
  "license": "ISC",
  "description": ""
}

# 本地运行

# 对照package.json的scripts
npm run dev
